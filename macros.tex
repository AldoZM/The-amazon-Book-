% \problema{titulo}{numLC}{rutaRelativaCpp}{dificultad}
% El título enlaza al .cpp local (vscode://, abre tu editor). Debajo, la animación
% del algoritmo con DOS accesos que cubren ambos escenarios:
%   local -> file://\repopath/... : repo descargado, funciona offline y sin Pages.
%   web   -> \pubbase/...         : hospedado en GitHub Pages, abre desde cualquier equipo online.
\newcommand{\problema}[4]{%
  \clearpage
  \phantomsection
  \addcontentsline{toc}{section}{#2. #1 \textnormal{(#4)}}%
  \section*{\href{vscode://file/\repopath/#3}{#2.\ #1}%
    \hfill\normalsize\textnormal{(#4)}}%
  \nobreak\noindent
  {\small\ensuremath{\triangleright}\ Ver animación\ \textperiodcentered\ %
    \href{file:///\repopath/rep-visual/problem.html?p=#2}{local}\ \textbar\ %
    \href{\pubbase/rep-visual/problem.html?p=#2}{web}}%
  \par\medskip
}
